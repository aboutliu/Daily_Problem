假设存在一颗权值线段树，你在树上查询 $op=2$ 或者 $op=3$ 。

- $op=2$，正常线段树二分。
- $op=3$ ，转化为最少多少个大于等于 $v+1$ ，再线段树二分。

怎么转化为线段树？

把所有的用得到的数字离线 $A$，建一颗长度 $|A|$ 的线段树。其实就是为每个数字分配一段区间，因为升序后相同数字会在这个区间。

然后更新就把数字放在这个区间。这样保证升序才可以正常二分。


```cpp 
#include <bits/stdc++.h>

using i64 = long long;
using u32 = unsigned;
using u64 = unsigned long long;

using i128 = __int128;
using u128 = unsigned __int128;

template<class Info>
struct SegmentTree {
#define ls i << 1
#define rs i << 1 | 1

    int N;
    std::vector<Info> seg;
    SegmentTree() {}
    SegmentTree(std::vector<int> a) {
        init(a);
    }
    void init(std::vector<int> a) {
        N = a.size();
        seg.resize(N + 1 << 2);

        auto build = [&](auto &&self, int i, int l, int r) -> void {
            seg[i] = {l, r};
            if (l == r) {
                seg[i].assign(a[l - 1]);
                return ;
            }
            int mid = (l + r) / 2;
            self(self, ls, l, mid);
            self(self, rs, mid + 1, r);
            pushup(i);
        };

        build(build, 1, 1, N);
    }
    void pushup(int i) {
        seg[i] = seg[ls] + seg[rs];
    }
    void modify(int i, int ind, std::pair<int, int> v) {
        if (seg[i].l == seg[i].r) {
            seg[i].apply(v);
            return ;
        }
        if (ind <= seg[ls].r) {
            modify(ls, ind, v);
        } else {
            modify(rs, ind, v);
        }
        pushup(i);
    }
    void modify(int p, std::pair<int, int> v) {
        modify(1, p, v);
    }
    Info rangeInfo(int i, int l, int r) {
        if (seg[i].l > r || seg[i].r < l) {
            return Info();
        }
        if (l <= seg[i].l && seg[i].r <= r) {
            return seg[i];
        }
        if (r <= seg[ls].r) {
            return rangeInfo(ls, l, r);
        } else if (l >= seg[rs].l) {
            return rangeInfo(rs, l, r);
        } else {
            return rangeInfo(ls, l, r) + rangeInfo(rs, l, r);
        }
    }
    Info rangeInfo(int l, int r) {
        return rangeInfo(1, l, r);
    }
    template<class F>
    int findFirst(int i, int l, int r, i64 v, F &&pred) {
        if (seg[i].l > r || seg[i].r < l || !pred(seg[i], v)) {
            return -1;
        }
        if (seg[i].l == seg[i].r) {
            return seg[i].l;
        }
        int res = findFirst(ls, l, r, v, pred);
        if (res == -1) {
            res = findFirst(rs, l, r, v - seg[ls].sum, pred);
        }
        return res;
    }
    template<class F>
    int findFirst(int l, int r, i64 v, F &&pred) {
        return findFirst(1, l, r, v, pred);
    }
    template<class F>
    int findLast(int i, int l, int r, i64 v, F &&pred) {
        if (seg[i].l > r || seg[i].r < l || !pred(seg[i], v)) {
            return -1;
        }
        if (seg[i].l == seg[i].r) {
            return seg[i].l;
        }
        int res = findLast(rs, l, r, v, pred);
        if (res == -1) {
            res = findLast(ls, l, r, v - seg[rs].sum, pred);
        }
        return res;
    }
    template<class F>
    int findLast(int l, int r, i64 v, F &&pred) {
        return findLast(1, l, r, v, pred);
    }
};

struct Info {
    int l, r;
    int sum;
    int cnt;
    void assign(int x) {
        sum = x;
        cnt = 0;
    }
    void apply(auto x) {
        sum = x.first;
        cnt = x.second;
    }
};

Info operator + (const Info &u, const Info &v) {
    Info res = {};
    res.l = u.l;
    res.r = v.r;
    res.sum = u.sum + v.sum;
    res.cnt = u.cnt + v.cnt;
    return res;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n;
    std::cin >> n;

    std::vector<int> a(n), ord;
    for (int i = 0; i < n; ++i) {
        std::cin >> a[i];
        ord.emplace_back(a[i]);
    }

    int q;
    std::cin >> q;

    std::vector<std::array<i64, 3>> ask(q);
    for (int i = 0; i < q; ++i) {
        i64 op;
        std::cin >> op;

        if (op == 1) {
            i64 x, y;
            std::cin >> x >> y;
            ask[i] = std::array {op, x, y};
            ord.emplace_back(y);
        } else {
            i64 v;
            std::cin >> v;
            ask[i] = std::array {op, v, 0LL};
        }
    }
    std::sort(ord.begin(), ord.end());

    const int N = 5E6;
    SegmentTree<Info> seg(std::vector<int>(ord.size() + 1, 0));

    std::vector<int> pos(N + 1, 1E9);
    for (int i = 0; i < ord.size(); ++i) {
        pos[ord[i]] = std::min(pos[ord[i]], i);
    }
    for (auto i : a) {
        seg.modify(pos[i] + 1, std::pair {i, 1});
        pos[i]++;
    }

    for (auto [op, x, y] : ask) {
        if (op == 1) {
            int &p = pos[a[x - 1]];
            seg.modify(p, std::pair {0, 0});
            p--;
            a[x - 1] = y;
            pos[y]++;
            seg.modify(pos[y], std::pair {y, 1});
        } else if (op == 2) {
            int p = seg.findLast(1, ord.size() + 1, x, [&](const auto &s, const i64 v) -> int {
                return s.sum >= v;
            });
            if (p == -1) {
                std::cout << p << "\n";
                continue;
            }
            int res = seg.rangeInfo(p, ord.size() + 1).cnt;
            std::cout << res << "\n";
        } else {
            int p = seg.findFirst(1, ord.size() + 1, x + 1, [&](const auto &s, const i64 v) -> int {
                return s.sum >= v;
            });
            if (p == -1) {
                std::cout << n << "\n";
                continue;
            }
            int res = seg.rangeInfo(1, p - 1).cnt;
            std::cout << res << "\n";
        }
    }

    return 0;
}
```

```cpp
#include <bits/stdc++.h>

using i64 = long long;
using u32 = unsigned;
using u64 = unsigned long long;

using i128 = __int128;
using u128 = unsigned __int128;

template <class T>
struct BitTrie {
    int id = 0, N;
    std::vector<std::array<int, 2>> trie {std::array<int, 2>()};
    std::vector<int> cnt {0};
    BitTrie() {}
    BitTrie(int _) {
        init(_);
    }
    void init(int _) {
        N = _;
        id = 0;
        cnt.assign(1, 0);
        trie.assign(1, std::array<int, 2>());
    }
    void insert(int x, int v) {
        int p = 0;
        for (int i = N - 1; i >= 0; --i) {
            int it = x >> i & 1;
            if (!trie[p][it]) {
                trie[p][it] = ++id;
                trie.push_back(std::array<int, 2>());
                cnt.push_back(0);
            }
            p = trie[p][it];
            cnt[p] += v;
        }
    }
    int find(int u, int x) {
        int p = 0, res = 0;
        for (int i = N - 1; i >= 0; --i) {
            int it = u >> i & 1, bx = x >> i & 1;
            if (!bx) {
                if (trie[p][it ^ 1]) {
                    res |= cnt[trie[p][it ^ 1]];
                }
                p = trie[p][it];
            } else {
                p = trie[p][it ^ 1];
            }
            if (!i) {
                res |= cnt[p];
            }
            if (!p) {
                break;
            }
        }
        return res;
    }
};

void solve() {
    int n, k;
    std::cin >> n >> k;

    std::vector<int> a(n);
    for (int i = 0; i < n; ++i) {
        std::cin >> a[i];
    }
    if (!k) {
        std::cout << "1\n";
        return;
    }

    int range = *max_element(a.begin(), a.end());
    BitTrie<int> bit(std::bit_width(1U * std::max(k, range)));

    int tans = 1E9;
    for (int i = n - 1, j = n - 1; i >= 0; --i) {
        bit.insert(a[i], 1);
        while (j >= i && bit.find(a[i], k) > 0) {
            tans = std::min(tans, j - i + 1);
            bit.insert(a[j], -1);
            j--;
        }
    }

    // bit.insert(3, 1);
    // std::cerr << bit.find(3, 3) << "\n";

    int ans = tans == 1E9 ? -1 : tans;
    std::cout << ans << "\n";
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int t;
    std::cin >> t;

    while (t--) {
        solve();
    }

    return 0;
}
```
